#include<iostream>
using namespace std;

int binary(int n) {
    int num = n; 
    int dec_value = 0; 
    int base = 1;  
    int temp = num; 
    while (temp>0) { 
        int last_digit = temp % 10; 
        temp = temp / 10; 
        dec_value += last_digit * base; 
        base = base * 2; 
    }    
	return dec_value;
}

int main() {
    int N, num; 
    cin>>N;
    for(int i=1;i<=N;i++){
        cin>>num;
        cout<< binary(num) <<endl;
    }
}
