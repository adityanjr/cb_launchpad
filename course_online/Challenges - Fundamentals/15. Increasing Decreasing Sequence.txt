
#include<iostream>
using namespace std;
int main(){
    int n, i;
    cin>>n;
    long int *arr = new long int[n];
    for (int i = 0; i < n; ++i){
        cin>>arr[i];
    }

    for(i = 0; i < n-1; i++){
        if (arr[i] <= arr[i + 1]){ 
            break;
        }
    }
    for (i += 1; i<n-1; ++i){
        if(arr[i] >= arr[i + 1]){    
            cout<<"false";
            return 0;
        }
    }
    cout<<"true";
    return 0;
}
