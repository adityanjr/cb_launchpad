#include<iostream>
using namespace std;
int main() {
    int n1, n2, x;  
    cin>>n1>>n2;
    if (n1>0 && n1<100 && n2>0 && n2<100){
        for(int i=1; i<=n1; i++){
            x=(3*i)+2;
            if( x%n2 != 0){
                cout<<x<<endl;
            } 
            else n1++;
        }
    }
	return 0;
}
